  /* ----
  [1] Generates font-size in rem, based on the given variable (i.e. $ms-up1)

      rhythm)

  ---- */
  font-size: $size; // [1]
  @if $lines != $base-line-multi { // [2]

  }

    line-height: ceil($size / $base-line-height) * ($base-line-height / $size);
  }
}
